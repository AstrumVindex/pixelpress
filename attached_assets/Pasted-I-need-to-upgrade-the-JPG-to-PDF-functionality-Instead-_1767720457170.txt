JavaScript

import React, { useState, useCallback, useEffect } from 'react';
import { Upload, X, FileImage, Check, Image as ImageIcon, Maximize, Minimize, AlignCenter, FileText } from 'lucide-react';

const JpgToPdfTool = () => {
  const [files, setFiles] = useState([]);
  const [previewUrls, setPreviewUrls] = useState({});
  const [isDragging, setIsDragging] = useState(false);
  const [isConverting, setIsConverting] = useState(false);
  const [isDone, setIsDone] = useState(false);
  const [resultFiles, setResultFiles] = useState([]);
  const [options, setOptions] = useState({
    orientation: 'portrait',
    pageSize: 'a4',
    margin: 'no',
    merge: true,
  });

  useEffect(() => {
    return () => Object.values(previewUrls).forEach(url => URL.revokeObjectURL(url));
  }, [previewUrls]);

  const handleFiles = useCallback((newFiles) => {
    const validFiles = Array.from(newFiles).filter(file => file.type === 'image/jpeg' || file.type === 'image/jpg');
    const newPreviews = {};
    validFiles.forEach(file => newPreviews[file.name] = URL.createObjectURL(file));
    setFiles(prev => [...prev, ...validFiles]);
    setPreviewUrls(prev => ({ ...prev, ...newPreviews }));
  }, []);

  const handleDrop = useCallback((e) => {
    e.preventDefault(); setIsDragging(false);
    if (e.dataTransfer.files?.length > 0) handleFiles(e.dataTransfer.files);
  }, [handleFiles]);

  const removeFile = useCallback((fileName) => {
    setFiles(prev => prev.filter(f => f.name !== fileName));
    setPreviewUrls(prev => {
      const newUrls = { ...prev };
      URL.revokeObjectURL(newUrls[fileName]);
      delete newUrls[fileName];
      return newUrls;
    });
  }, []);

  const handleConvert = () => {
    if (files.length === 0) return;
    setIsConverting(true);
    setTimeout(() => {
      setIsConverting(false);
      setIsDone(true);
      if (options.merge) {
        setResultFiles([{ name: 'merged-document.pdf', size: 1024 * 1024 * 2.5 }]);
      } else {
        setResultFiles(files.map(file => ({
          name: file.name.replace(/\.(jpg|jpeg)$/i, '.pdf'),
          size: file.size * 1.2
        })));
      }
    }, 2000);
  };

  if (isDone) {
    return (
      <div className="max-w-3xl mx-auto p-8 bg-white rounded-2xl shadow-sm">
        <div className="text-center mb-8">
          <div className="w-16 h-16 bg-green-100 rounded-full flex items-center justify-center mx-auto mb-4"><Check className="w-8 h-8 text-green-600" /></div>
          <h2 className="text-2xl font-bold text-slate-900">Conversion Complete!</h2>
        </div>
        <div className="space-y-3 mb-8">
          {resultFiles.map((file, i) => (
            <div key={i} className="flex items-center justify-between p-4 bg-slate-50 rounded-xl border">
              <div className="flex items-center gap-4"><FileText className="text-red-500" /><p className="font-medium">{file.name}</p></div>
              <button className="px-6 py-2 bg-indigo-600 text-white rounded-lg">Download</button>
            </div>
          ))}
        </div>
        <button onClick={() => { setIsDone(false); setFiles([]); }} className="w-full py-3 bg-slate-100 font-medium rounded-lg">Convert More</button>
      </div>
    );
  }

  return (
    <div className="flex flex-col lg:flex-row gap-8">
      <div className="flex-1">
        {files.length === 0 ? (
          <div className={`border-2 border-dashed rounded-3xl p-12 text-center h-96 flex flex-col items-center justify-center ${isDragging ? 'border-indigo-500 bg-indigo-50' : 'border-slate-200'}`}
            onDragOver={(e) => { e.preventDefault(); setIsDragging(true); }}
            onDragLeave={(e) => { e.preventDefault(); setIsDragging(false); }}
            onDrop={handleDrop}>
            <Upload className="w-12 h-12 text-indigo-600 mb-4" />
            <h3 className="text-xl font-bold text-slate-900">Upload JPG images</h3>
            <input type="file" id="jpg-upload" className="hidden" multiple accept="image/jpeg, image/jpg" onChange={(e) => handleFiles(e.target.files)} />
            <label htmlFor="jpg-upload" className="mt-6 px-8 py-3 bg-indigo-600 text-white font-semibold rounded-full cursor-pointer">Select Files</label>
          </div>
        ) : (
          <div className="bg-slate-50 p-6 rounded-3xl min-h-[24rem]">
            <div className="grid grid-cols-2 sm:grid-cols-3 gap-4">
              {files.map(file => (
                <div key={file.name} className="relative group rounded-xl overflow-hidden shadow-sm bg-white">
                  <img src={previewUrls[file.name]} alt="" className="w-full h-40 object-cover" />
                  <button onClick={() => removeFile(file.name)} className="absolute top-2 right-2 p-1 bg-white rounded-full text-red-500 shadow-sm"><X size={16} /></button>
                </div>
              ))}
              <label className="border-2 border-dashed border-slate-300 rounded-xl h-40 flex flex-col items-center justify-center cursor-pointer hover:bg-slate-100">
                <Upload className="w-6 h-6 text-slate-400" /><span className="text-xs text-slate-500 mt-2">Add More</span>
                <input type="file" className="hidden" multiple accept="image/jpeg, image/jpg" onChange={(e) => handleFiles(e.target.files)} />
              </label>
            </div>
          </div>
        )}
      </div>

      <div className="w-full lg:w-80 bg-white p-6 rounded-3xl shadow-sm h-fit">
        <h3 className="text-lg font-bold text-slate-900 mb-6">PDF Options</h3>
        <div className="mb-6">
          <label className="text-sm font-medium text-slate-700 block mb-2">Orientation</label>
          <div className="grid grid-cols-2 gap-2">
            <button onClick={() => setOptions({...options, orientation: 'portrait'})} className={`p-2 rounded-lg border text-sm ${options.orientation === 'portrait' ? 'border-indigo-600 bg-indigo-50 text-indigo-600' : 'border-slate-200'}`}>Portrait</button>
            <button onClick={() => setOptions({...options, orientation: 'landscape'})} className={`p-2 rounded-lg border text-sm ${options.orientation === 'landscape' ? 'border-indigo-600 bg-indigo-50 text-indigo-600' : 'border-slate-200'}`}>Landscape</button>
          </div>
        </div>
        <div className="mb-6">
          <label className="text-sm font-medium text-slate-700 block mb-2">Margin</label>
          <div className="grid grid-cols-3 gap-2">
            {['no', 'small', 'big'].map(m => (
              <button key={m} onClick={() => setOptions({...options, margin: m})} className={`p-2 rounded-lg border text-xs capitalize ${options.margin === m ? 'border-indigo-600 bg-indigo-50 text-indigo-600' : 'border-slate-200'}`}>{m}</button>
            ))}
          </div>
        </div>
        <label className="flex items-center gap-3 p-3 border rounded-xl cursor-pointer mb-6">
          <input type="checkbox" checked={options.merge} onChange={(e) => setOptions({...options, merge: e.target.checked})} className="accent-indigo-600 w-5 h-5" />
          <span className="text-sm font-medium text-slate-700">Merge all into one PDF</span>
        </label>
        <button onClick={handleConvert} disabled={files.length===0 || isConverting} className="w-full py-4 bg-indigo-600 text-white rounded-xl font-bold disabled:bg-slate-300">
          {isConverting ? 'Creating PDF...' : 'Convert to PDF'}
        </button>
      </div>
    </div>
  );
};
export default JpgToPdfTool;
